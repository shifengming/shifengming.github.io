# axios的用法

# 安装axios

npm install axios -S

# 创建fetch.js文件，封装axios

import axios from 'axios';

// 自动切换环境
if(process.env.NODE_ENV == 'development') {
    axios.defaults.baseURL = '/sso';//  本地开发环境
} else if(process.env.NODE_ENV == 'debug') {
    axios.defaults.baseURL = '/sso';  // 本地调试环境
} else if(process.env.NODE_ENV == 'production') {
    axios.defaults.baseURL = localStorage.getItem('domain'); // 线上环境
}
axios.defaults.baseURL ='https://api.apiopen.top'
// 设置超时时间
axios.defaults.timeout = 10000;

// 设置请求头
axios.defaults.headers.post['Content-Type'] = 'application/json';
axios.defaults.headers.get['Content-Type'] = 'application/x-www-form-urlencoded';
// axios.defaults.withCredentials=true; // 允许跨域

// 对外请求接口 ===》export导出request方法
export default function request({method, url, params}) {
    if(method == 'GET') {
        return get(url, params)
    } else if(method == 'POST') {
        return post(url, params)
    }
}

// 封装对内的get请求
function get(url, params) {
    return new Promise((resolve, reject) => {
        axios.get(url, params).then(res => {
            resolve(res.data)
        }).catch(err => {
            reject(err)
        })
    })
}

// 封装对内的post请求
function post(url, params) {
    return new Promise((resolve, reject) => {
        axios.post(url, JSON.stringify(params)).then(res => {
            resolve(res.data)
        }).catch(err => {
            reject(err)
        })
    })
}

// 请求拦截（请求发出前处理请求）
axios.interceptors.request.use(
    config => {
        if(localStorage.getItem('sessionId')) {
            config.headers.sessionId = localStorage.getItem('sessionId')
        }
        return config
    },
    error => {
        return Promise.error(error)
    }
)

// 响应拦截器（处理响应数据）
axios.interceptors.response.use(
    response =>{
        if(response.status == 200) {
            return Promise.resolve(response)
        } else {
            return Promise.reject(response)
        }
    },
    error => {
        return Promise.reject(error.response)
    }
)

# 在app.vue里面使用

当然也可以在别的位置使用，可能会涉及父子组件传递的问题

<template>
  <div id="app">
    <img alt="Vue logo" src="./assets/logo.png">
    <Login msg="Welcome to Your Vue.js App"/>
    <button>登录</button>
    
    <el-table :data="list" border stripe>
      <el-table-column type="index"></el-table-column>
      <el-table-column label="姓名" prop="name"></el-table-column>
      <el-table-column label="时间" prop="passtime"></el-table-column>
      <el-table-column label="序号" prop="sid"></el-table-column>
      <el-table-column label="内容" prop="text"></el-table-column>
      <el-table-column label="时间" prop="passtime"></el-table-column>
   </el-table>

  </div>
</template>

<script>
import Login from './components/Login.vue'
import { login } from './components/api'; // 导入刚才login方法

export default {
    data(){
      return{
        list:[]
      }
  },
   created() {
       this.login()
   },
  // name: 'App',
 methods: {
      login() {
        this.name = []  // 初始化数据

                  console.log('暂无该项数据')
                }
        })
    }
  },
  components: {
    Login
  }
}
</script>

# 设置跨域

注意：一般接口都是不允许跨域访问的，我们需要在webpack配置文件中设置跨域

在config/index.js里加入：

proxyTable: {
      '/sso': {
        target: 'http://192.168.0.225:22100',  // 通过本地服务器将你的请求转发到这个地址
        changeOrigin: true,  // 设置这个参数可以避免跨域
        pathRewrite: {
          '^/sso': '/sso'
        }
      }

# params和query传参的区别

  • query方式的参数会在地址栏后显示出来
  • params方式的参数不会在地址栏后显示出来
  • query可以用name或path来引入
  • params必需要用name来引入
接收参数都是类似的，分别是:this.$route.query.name和this.$route.params.name
